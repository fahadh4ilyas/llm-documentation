import httpx
import time


NUM_TRIAL = 3
BASE_URL = "http://localhost:8000/v1" # Masukkan url rest API vLLM dengan endpoint /v1
SYSTEM_PROMPT = "You are an AI agent" # ganti dengan system prompt yang ingin kalian gunakan


def get_model():
    trial = 0
    while trial < NUM_TRIAL:
        with httpx.Client() as client:
            response = client.get(BASE_URL + '/models')
            try:
                response.raise_for_status()
                return response.json()['data'][0]['id']
            except httpx.HTTPStatusError as exc:
                print(f"Error response {exc.response.status_code} while requesting {exc.request.url}")
        trial += 1
        if trial < NUM_TRIAL:
            print(f'Retrying in {2**(trial-1)} second(s)...')
            time.sleep(2**(trial-1))


def chat_completions(messages, model):
    trial = 0
    while trial < NUM_TRIAL:
        with httpx.Client(headers={"Authorization": "Bearer EMPTY"}) as client:
            response = client.post(BASE_URL + '/chat/completions', json={
                "messages": messages,
                "model": model
            })
            try:
                response.raise_for_status()
                return response.json()['choices'][0]['message']
            except httpx.HTTPStatusError as exc:
                print(f"Error response {exc.response.status_code} while requesting {exc.request.url}")
        trial += 1
        if trial < NUM_TRIAL:
            print(f'Retrying in {2**(trial-1)} second(s)...')
            time.sleep(2**(trial-1))


def main():
    model = get_model()
    if not model:
        print('Something wrong with connection to vLLM. Are you sure it\'s up?')
        return


    messages = [
            {
                "role": "system",
                "content": SYSTEM_PROMPT
            }
        ]
    while True:
        user_message = input('(Ctrl+C to stop) Human: ')
        messages.append(
            {
                "role": "user",
                "content": user_message
            }
        )


        assistant_message = chat_completions(messages, model)
        if assistant_message.get('reasoning', None):
            print('AI thinking:', assistant_message['reasoning'].strip())
        print('AI:', assistant_message['content'].strip())
        messages.append(assistant_message)


if __name__ == '__main__':
    try:
        main()
    except KeyboardInterrupt:
        print('\nGood bye!')

import openai
import time


NUM_TRIAL = 3
BASE_URL = "http://localhost:8000/v1" # Masukkan url rest API vLLM dengan endpoint /v1
SYSTEM_PROMPT = "You are an AI agent" # ganti dengan system prompt yang ingin kalian gunakan


def get_model():
    trial = 0
    while trial < NUM_TRIAL:
        with openai.OpenAI(base_url=BASE_URL, api_key='EMPTY') as client:
            try:
                response = client.models.list()
                return response.data[0].id
            except openai.OpenAIError as exc:
                print(f"Error trying to access model: {exc}")
        trial += 1
        if trial < NUM_TRIAL:
            print(f'Retrying in {2**(trial-1)} second(s)...')
            time.sleep(2**(trial-1))


def chat_completions(messages, model):
    trial = 0
    while trial < NUM_TRIAL:
        client = openai.OpenAI(base_url=BASE_URL, api_key='EMPTY')
        try:
            response = client.chat.completions.create(
                messages=messages,
                model=model,
                stream=True
            )
            return client, response
        except openai.OpenAIError as exc:
            print(f"Error trying to create chat completions: {exc}")
            client.close()
        trial += 1
        if trial < NUM_TRIAL:
            print(f'Retrying in {2**(trial-1)} second(s)...')
            time.sleep(2**(trial-1))


def main():
    model = get_model()
    if not model:
        print('Something wrong with connection to vLLM. Are you sure it\'s up?')
        return


    messages = [
            {
                "role": "system",
                "content": SYSTEM_PROMPT
            }
        ]
    while True:
        user_message = input('(Ctrl+C to stop) Human: ')
        messages.append(
            {
                "role": "user",
                "content": user_message
            }
        )


        client, response = chat_completions(messages, model)
        assistant_message = {
            "role": "assistant",
            "reasoning": "",
            "content": ""
        }
        printed_begin_reasoning = False
        printed_begin_content = False
        for chunk in response:
            reasoning = getattr(chunk.choices[0].delta, 'reasoning', None)
            content = getattr(chunk.choices[0].delta, 'content', None)
            if reasoning is not None:
                assistant_message['reasoning'] += reasoning
            if content is not None:
                assistant_message['content'] += content
            if reasoning:
                if not printed_begin_reasoning:
                    printed_begin_reasoning = True
                    print('AI thinking: ', end="", flush=True)
                    reasoning = reasoning.lstrip()
                print(reasoning, end="", flush=True)
            if content:
                if not printed_begin_content:
                    printed_begin_content = True
                    print('AI: ', end="", flush=True)
                    content = content.lstrip()
                print(content, end="", flush=True)
        print()
        client.close()
        messages.append(assistant_message)


if __name__ == '__main__':
    try:
        main()
    except KeyboardInterrupt:
        print('\nGood bye!')

1.  **Analyze the Request:**
    *   Input: "Hello!"
    *   Intent: Greeting.
    *   Role: AI agent (as per system instruction).
    *   Expected Output: A friendly, appropriate response to the greeting.


2.  **Determine the Tone:**
    *   Friendly, helpful, polite, engaging.
    *   Avoid overly formal or robotic language, but maintain professionalism.


3.  **Draft Potential Responses:**
    *   "Hello! How can I help you today?"
    *   "Hi there! What's up?"
    *   "Hello! Nice to meet you. What do you need?"
    *   "Hey! How's it going?"


4.  **Select the Best Response:**
    *   "Hello! How can I help you today?" is standard and effective for an AI assistant.
    *   Adding a bit of warmth makes it better. "Hello! 👋 How are you doing today? Is there anything I can help you with?"


5.  **Final Polish:**
    *   Keep it concise.
    *   Ensure it aligns with the "AI agent" persona.


6.  **Final Output Generation:** (Matches the selected draft)
    "Hello! 👋 How can I help you today?" or similar.
